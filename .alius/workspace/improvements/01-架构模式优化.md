# 01 - 架构模式优化

## 📋 概述

在保持 Alius 核心 Rust 架构的基础上，借鉴竞品的项目组织模式，优化模块划分、依赖管理和构建系统。

**核心原则：保持 Rust Cargo workspace，优化模块边界，提升可维护性**

## 🎯 目标

1. 优化 Cargo workspace 结构
2. 明确模块职责和依赖关系
3. 引入插件系统支持动态扩展
4. 改进构建和发布流程
5. 提升代码复用性

## 📊 现状分析

### Alius 当前架构

```
alius_cli/
├── entrypoints/
│   └── cli/          # CLI 入口
├── protocol/         # 协议定义
├── runtime/
```

**优势：**
- ✅ 清晰的分层架构
- ✅ runtime 和 entrypoint 分离
- ✅ Cargo workspace 管理依赖

**改进空间：**
- ❌ 缺少插件系统
- ❌ 模块间依赖关系可以更清晰
- ❌ 缺少 SDK 层供外部集成
- ❌ 没有明确的扩展点

### 3. 插件系统设计

```rust
// crates/alius-plugin/src/lib.rs
use async_trait::async_trait;
use serde::{Deserialize, Serialize};

/// 插件元数据
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct PluginMetadata {
    pub name: String,
    pub version: String,
    pub author: String,
    pub description: String,
    pub homepage: Option<String>,
    pub license: String,
    pub dependencies: Vec<String>,
}

/// 插件接口
#[async_trait]
pub trait Plugin: Send + Sync {
    /// 插件元数据
    fn metadata(&self) -> &PluginMetadata;
    
    /// 插件初始化
    async fn initialize(&mut self, context: &PluginContext) -> anyhow::Result<()>;
    
    /// 插件卸载
    async fn shutdown(&mut self) -> anyhow::Result<()>;
    
    /// 注册插件提供的工具
    fn register_tools(&self) -> Vec<Box<dyn alius_tools::AliusTool>> {
        Vec::new()
    }
    
    /// 注册插件提供的命令
    fn register_commands(&self) -> Vec<Box<dyn Command>> {
        Vec::new()
    }
    
    /// 注册插件提供的钩子
    fn register_hooks(&self) -> Vec<Box<dyn Hook>> {
        Vec::new()
    }
}

/// 插件上下文
pub struct PluginContext {
    pub config_dir: PathBuf,
    pub data_dir: PathBuf,
    pub cache_dir: PathBuf,
    pub runtime_config: Arc<alius_config::Settings>,
}

/// 插件加载器
pub struct PluginLoader {
    plugins: HashMap<String, Box<dyn Plugin>>,
    plugin_dir: PathBuf,
}

impl PluginLoader {
    pub fn new(plugin_dir: PathBuf) -> Self {
        Self {
            plugins: HashMap::new(),
            plugin_dir,
        }
    }
    
    /// 从目录加载所有插件
    pub async fn load_all(&mut self, context: &PluginContext) -> anyhow::Result<()> {
        let entries = std::fs::read_dir(&self.plugin_dir)?;
        
        for entry in entries {
            let entry = entry?;
            let path = entry.path();
            
            if path.is_dir() {
                match self.load_plugin(&path, context).await {
                    Ok(plugin) => {
                        let name = plugin.metadata().name.clone();
                        self.plugins.insert(name.clone(), plugin);
                        log::info!("Loaded plugin: {}", name);
                    }
                    Err(e) => {
                        log::warn!("Failed to load plugin from {:?}: {}", path, e);
                    }
                }
            }
        }
        
        Ok(())
    }
    
    /// 加载单个插件
    async fn load_plugin(
        &self,
        path: &Path,
        context: &PluginContext,
    ) -> anyhow::Result<Box<dyn Plugin>> {
        // 读取插件元数据
        let manifest_path = path.join("plugin.toml");
        let manifest_content = std::fs::read_to_string(manifest_path)?;
        let metadata: PluginMetadata = toml::from_str(&manifest_content)?;
        
        // 动态加载插件库
        // 方案 1: 使用 libloading 动态加载 .so/.dylib/.dll
        // 方案 2: 使用 WASM 插件
        // 方案 3: 使用进程隔离的插件（通过 stdio 通信）
        
        // 这里使用方案 3（最安全）
        let plugin = StdioPlugin::new(path, metadata)?;
        let mut boxed_plugin: Box<dyn Plugin> = Box::new(plugin);
        boxed_plugin.initialize(context).await?;
        
        Ok(boxed_plugin)
    }
    
    /// 获取插件
    pub fn get(&self, name: &str) -> Option<&dyn Plugin> {
        self.plugins.get(name).map(|p| p.as_ref())
    }
    
    /// 列出所有插件
    pub fn list(&self) -> Vec<&PluginMetadata> {
        self.plugins.values().map(|p| p.metadata()).collect()
    }
}
```

### 4. 钩子系统

```rust
// crates/alius-hooks/src/lib.rs
use async_trait::async_trait;
use serde_json::Value;

/// 钩子类型
#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
pub enum HookType {
    // 生命周期钩子
    BeforeStart,
    AfterStart,
    BeforeShutdown,
    AfterShutdown,
    
    // 会话钩子
    BeforeSession,
    AfterSession,
    
    // 消息钩子
    BeforeMessage,
    AfterMessage,
    
    // 工具钩子
    BeforeToolCall,
    AfterToolCall,
    
    // 配置钩子
    BeforeConfigLoad,
    AfterConfigSave,
}

/// 钩子上下文
#[derive(Debug, Clone)]
pub struct HookContext {
    pub hook_type: HookType,
    pub data: Value,
}

/// 钩子结果
#[derive(Debug, Clone)]
pub enum HookResult {
    Continue,
    Modified(Value),
    Abort(String),
}

/// 钩子接口
#[async_trait]
pub trait Hook: Send + Sync {
    fn name(&self) -> &str;
    fn hook_type(&self) -> HookType;
    async fn execute(&self, context: &HookContext) -> anyhow::Result<HookResult>;
}

/// 钩子注册表
pub struct HookRegistry {
    hooks: HashMap<HookType, Vec<Box<dyn Hook>>>,
}

impl HookRegistry {
    pub fn new() -> Self {
        Self {
            hooks: HashMap::new(),
        }
    }
    
    /// 注册钩子
    pub fn register(&mut self, hook: Box<dyn Hook>) {
        let hook_type = hook.hook_type();
        self.hooks.entry(hook_type).or_insert_with(Vec::new).push(hook);
    }
    
    /// 执行钩子
    pub async fn execute(&self, hook_type: HookType, data: Value) -> anyhow::Result<Value> {
        let context = HookContext { hook_type, data };
        let mut current_data = context.data.clone();
        
        if let Some(hooks) = self.hooks.get(&hook_type) {
            for hook in hooks {
                let ctx = HookContext {
                    hook_type,
                    data: current_data.clone(),
                };
                
                match hook.execute(&ctx).await? {
                    HookResult::Continue => {},
                    HookResult::Modified(new_data) => {
                        current_data = new_data;
                    }
                    HookResult::Abort(reason) => {
                        anyhow::bail!("Hook aborted: {}", reason);
                    }
                }
            }
        }
        
        Ok(current_data)
    }
}
```

### 5. 扩展示例：Web Search

```rust
// extensions/alius-ext-web-search/src/lib.rs
use alius_plugin::{Plugin, PluginMetadata, PluginContext};
use alius_tools::{AliusTool, ToolContext, ToolResult};
use async_trait::async_trait;

pub struct WebSearchExtension {
    metadata: PluginMetadata,
}

impl WebSearchExtension {
    pub fn new() -> Self {
        Self {
            metadata: PluginMetadata {
                name: "web-search".to_string(),
                version: env!("CARGO_PKG_VERSION").to_string(),
                author: "Alius Team".to_string(),
                description: "Web search capabilities".to_string(),
                homepage: Some("https://github.com/alius/extensions/web-search".to_string()),
                license: "MIT".to_string(),
                dependencies: vec![],
            },
        }
    }
}

#[async_trait]
impl Plugin for WebSearchExtension {
    fn metadata(&self) -> &PluginMetadata {
        &self.metadata
    }
    
    async fn initialize(&mut self, _context: &PluginContext) -> anyhow::Result<()> {
        log::info!("Web search extension initialized");
        Ok(())
    }
    
    async fn shutdown(&mut self) -> anyhow::Result<()> {
        log::info!("Web search extension shutdown");
        Ok(())
    }
    
    fn register_tools(&self) -> Vec<Box<dyn AliusTool>> {
        vec![
            Box::new(WebSearchTool::new()),
            Box::new(WebFetchTool::new()),
        ]
    }
}

// 导出插件创建函数
#[no_mangle]
pub extern "C" fn create_plugin() -> *mut dyn Plugin {
    Box::into_raw(Box::new(WebSearchExtension::new()))
}
```

### 6. CLI 集成插件系统

```rust
// crates/alius-cli/src/main.rs
use alius_plugin::{PluginLoader, PluginContext};
use alius_hooks::HookRegistry;

#[tokio::main]
async fn main() -> anyhow::Result<()> {
    // 初始化插件系统
    let plugin_dir = dirs::home_dir()
        .unwrap()
        .join(".alius/plugins");
    
    let mut plugin_loader = PluginLoader::new(plugin_dir);
    let plugin_context = PluginContext {
        config_dir: dirs::config_dir().unwrap().join("alius"),
        data_dir: dirs::data_dir().unwrap().join("alius"),
        cache_dir: dirs::cache_dir().unwrap().join("alius"),
        runtime_config: Arc::new(settings),
    };
    
    plugin_loader.load_all(&plugin_context).await?;
    
    // 注册插件提供的工具和钩子
    let mut tool_registry = alius_tools::ToolRegistry::new();
    let mut hook_registry = HookRegistry::new();
    
    for plugin in plugin_loader.list() {
        let plugin_instance = plugin_loader.get(&plugin.name).unwrap();
        
        for tool in plugin_instance.register_tools() {
            tool_registry.register(tool)?;
        }
        
        for hook in plugin_instance.register_hooks() {
            hook_registry.register(hook);
        }
    }
    
    // 继续启动应用
    // ...
}
```
