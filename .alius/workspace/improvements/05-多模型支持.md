# 05 - 多模型支持

## 📋 概述

扩展 Alius 的模型支持能力，实现统一的模型提供商抽象层，支持 OpenAI、Anthropic、Google、AWS Bedrock、Azure、本地模型等多种来源。

## 🎯 目标

1. 抽象统一的模型提供商接口
2. 支持主流云服务商（AWS Bedrock、Google Vertex AI、Azure OpenAI）
3. 支持本地模型（Ollama、LM Studio、vLLM）
4. 实现智能模型路由和降级策略
5. 提供模型池管理和负载均衡

## 📊 现状分析

### Alius 当前状态
- ✅ 支持 OpenAI、Anthropic、BigModel、DeepSeek
- ✅ 有基础的模型提供商抽象（ModelProvider trait）
- ❌ 缺少 AWS Bedrock、Google Vertex AI 支持
- ❌ 缺少本地模型集成
- ❌ 没有模型路由策略
- ❌ 配置较为分散

```rust
// 模型路由器
pub struct ModelRouter {
    primary: Box<dyn ModelProvider>,
    fallbacks: Vec<Box<dyn ModelProvider>>,
    retry_config: RetryConfig,
}

impl ModelRouter {
    pub async fn complete(&self, request: CompletionRequest) -> Result<CompletionResponse> {
        let mut last_error = None;
        
        // 尝试主模型
        match self.primary.complete(request.clone()).await {
            Ok(response) => return Ok(response),
            Err(e) => last_error = Some(e),
        }
        
        // 降级到备用模型
        for fallback in &self.fallbacks {
            match fallback.complete(request.clone()).await {
                Ok(response) => return Ok(response),
                Err(e) => last_error = Some(e),
            }
        }
        
        Err(last_error.unwrap())
    }
}
```

```rust
// runtime/model/src/providers/bedrock.rs
use aws_config::BehaviorVersion;
use aws_sdk_bedrock_runtime::{Client, primitives::Blob};
use anthropic_bedrock_sdk::AnthropicBedrock;

pub struct BedrockProvider {
    metadata: ProviderMetadata,
    client: Option<AnthropicBedrock>,
    region: String,
}

impl BedrockProvider {
    pub fn new() -> Self {
        Self {
            metadata: ProviderMetadata {
                id: "bedrock".to_string(),
                name: "AWS Bedrock".to_string(),
                description: "Amazon Bedrock managed AI models".to_string(),
                supported_capabilities: vec![
                    Capability::TextGeneration,
                    Capability::Streaming,
                    Capability::FunctionCalling,
                    Capability::Vision,
                ],
                requires_api_key: false,
                requires_region: true,
                default_region: Some("us-east-1".to_string()),
            },
            client: None,
            region: "us-east-1".to_string(),
        }
    }
}

#[async_trait]
impl ModelProvider for BedrockProvider {
    fn metadata(&self) -> &ProviderMetadata {
        &self.metadata
    }
    
    async fn initialize(&mut self, config: ProviderConfig) -> anyhow::Result<()> {
        self.region = config.region.unwrap_or_else(|| "us-east-1".to_string());
        
        // 使用 AWS SDK 配置
        let aws_config = aws_config::load_defaults(BehaviorVersion::latest()).await;
        
        self.client = Some(AnthropicBedrock::new(
            aws_config.region().map(|r| r.to_string()).unwrap_or(self.region.clone()),
            config.credentials_profile,
        )?);
        
        Ok(())
    }
    
    async fn list_models(&self) -> anyhow::Result<Vec<ModelDefinition>> {
        Ok(vec![
            ModelDefinition {
                id: "anthropic.claude-3-5-sonnet-20241022-v2:0".to_string(),
                name: "Claude 3.5 Sonnet v2".to_string(),
                provider: "bedrock".to_string(),
                context_window: 200_000,
                max_output_tokens: 8_192,
                supports_streaming: true,
                supports_functions: true,
                supports_vision: true,
                cost_per_million_input: 3.0,
                cost_per_million_output: 15.0,
            },
            // ... 其他模型
        ])
    }
    
    async fn create_completion(
        &self,
        model: &str,
        messages: Vec<Message>,
        options: CompletionOptions,
    ) -> anyhow::Result<CompletionResponse> {
        let client = self.client.as_ref()
            .ok_or_else(|| anyhow::anyhow!("Bedrock client not initialized"))?;
        
        let response = client.messages()
            .model(model)
            .messages(messages.into_iter().map(|m| m.into()).collect())
            .max_tokens(options.max_tokens.unwrap_or(4096))
            .temperature(options.temperature.unwrap_or(1.0))
            .send()
            .await?;
        
        Ok(response.into())
    }
    
    async fn create_streaming_completion(
        &self,
        model: &str,
        messages: Vec<Message>,
        options: CompletionOptions,
    ) -> anyhow::Result<impl Stream<Item = Result<StreamChunk>>> {
        let client = self.client.as_ref()
            .ok_or_else(|| anyhow::anyhow!("Bedrock client not initialized"))?;
        
        let stream = client.messages()
            .model(model)
            .messages(messages.into_iter().map(|m| m.into()).collect())
            .max_tokens(options.max_tokens.unwrap_or(4096))
            .stream()
            .await?;
        
        Ok(stream.map(|chunk| chunk.map(|c| c.into())))
    }
    
    async fn health_check(&self) -> anyhow::Result<HealthStatus> {
        // 简单的连接测试
        match self.list_models().await {
            Ok(_) => Ok(HealthStatus::Healthy),
            Err(e) => Ok(HealthStatus::Unhealthy(e.to_string())),
        }
    }
}
```

### 5. 模型路由器

```rust
// runtime/model/src/router.rs
use std::sync::Arc;
use tokio::sync::RwLock;

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct RoutingRule {
    pub name: String,
    pub condition: RoutingCondition,
    pub target_provider: String,
    pub target_model: String,
    pub priority: u32,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum RoutingCondition {
    /// 总是匹配
    Always,
    /// 基于模型 ID
    ModelId(String),
    /// 基于能力要求
    RequiresCapability(Capability),
    /// 基于成本限制
    CostLimit { max_cost_per_million: f64 },
    /// 自定义表达式
    Expression(String),
}

pub struct ModelRouter {
    providers: Arc<RwLock<HashMap<String, Box<dyn ModelProvider>>>>,
    rules: Vec<RoutingRule>,
    default_provider: String,
    default_model: String,
}

impl ModelRouter {
    pub fn new(default_provider: String, default_model: String) -> Self {
        Self {
            providers: Arc::new(RwLock::new(HashMap::new())),
            rules: Vec::new(),
            default_provider,
            default_model,
        }
    }
    
    pub async fn register_provider(
        &self,
        id: String,
        provider: Box<dyn ModelProvider>,
    ) -> anyhow::Result<()> {
        self.providers.write().await.insert(id, provider);
        Ok(())
    }
    
    pub fn add_rule(&mut self, rule: RoutingRule) {
        self.rules.push(rule);
        self.rules.sort_by_key(|r| r.priority);
    }
    
    pub async fn route(
        &self,
        request: &CompletionRequest,
    ) -> anyhow::Result<(String, String)> {
        // 按优先级检查路由规则
        for rule in &self.rules {
            if self.matches_condition(&rule.condition, request).await? {
                return Ok((rule.target_provider.clone(), rule.target_model.clone()));
            }
        }
        
        // 默认路由
        Ok((self.default_provider.clone(), self.default_model.clone()))
    }
    
    async fn matches_condition(
        &self,
        condition: &RoutingCondition,
        request: &CompletionRequest,
    ) -> anyhow::Result<bool> {
        match condition {
            RoutingCondition::Always => Ok(true),
            RoutingCondition::ModelId(id) => {
                Ok(request.model.as_deref() == Some(id.as_str()))
            }
            RoutingCondition::RequiresCapability(cap) => {
                // 检查请求是否需要特定能力
                Ok(match cap {
                    Capability::Vision => request.has_images(),
                    Capability::FunctionCalling => request.tools.is_some(),
                    _ => false,
                })
            }
            RoutingCondition::CostLimit { max_cost_per_million } => {
                // 基于成本限制路由
                Ok(true) // 简化实现
            }
            RoutingCondition::Expression(_) => {
                // 自定义表达式评估
                Ok(false) // TODO: 实现表达式引擎
            }
        }
    }
    
    pub async fn complete_with_routing(
        &self,
        mut request: CompletionRequest,
    ) -> anyhow::Result<CompletionResponse> {
        let (provider_id, model_id) = self.route(&request).await?;
        request.model = Some(model_id.clone());
        
        let providers = self.providers.read().await;
        let provider = providers.get(&provider_id)
            .ok_or_else(|| anyhow::anyhow!("Provider not found: {}", provider_id))?;
        
        provider.create_completion(
            &model_id,
            request.messages,
            request.options,
        ).await
    }
}
```
